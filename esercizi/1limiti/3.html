\[ \lim_{x \to +\infty} \dfrac{2x^2 - 3x + 1}{6x - 20} \]
Soluzione
\[ \begin{align*} \lim_{x \to +\infty} \dfrac{2x^2 - 3x + 1}{6x - 20} = \\\\ \lim_{x \to +\infty} \dfrac{2x^2}{6x} = \\\\ \lim_{x \to +\infty} \dfrac{x}{3} = \dfrac{+\infty}{3} = +\infty \end{align*} \]

\[ \lim_{x \to -\infty} \dfrac{- 3x^3 + 3x^2 - 1}{2x^4 - x^2 +1} \]
Soluzione
\[ \begin{align*} \lim_{x \to -\infty} \dfrac{- 3x^3 + 3x^2 - 1}{2x^4 - x^2 +1} \\\\ \lim_{x \to +\infty} \dfrac{- 3x^3}{2x^4} = \\\\ \lim_{x \to +\infty} \dfrac{-3}{2x} = \dfrac{-3}{2 \cdot (+\infty)} \\\\ \dfrac{-3}{+\infty} = 0 \end{align*} \]

\[ \lim_{x \to -\infty} \dfrac{6x - 4}{3x - 5} \]
Soluzione
\[ \begin{align*} \lim_{x \to -\infty} \dfrac{6x - 4}{3x - 5} = \\\\ \lim_{x \to -\infty} \dfrac{6x}{3x} = 2 \end{align*} \]
