\[ \begin{align*} f''(x) &= \left[f'(x)\right]' = \left[14x\,sin(x) + 7x^2\,cos(x)\right]' = \\\\ &= \left[14x\,sin(x)]' + [7x^2\,cos(x)\right]' = \\\\ &= 14\,\left[x\,sin(x)\right]' + 7\left[x^2\,cos(x)\right]' = \\\\ &= 14\,\left[sin(x) + x\,cos(x)\right] + 7\,\left[2x\,cos(x) + x^2\,(-sin(x))\right] \\\\ &= 14\,sin(x) + 14\,x\,cos(x) + 14\,x\,cos(x) -7\,x^2\,sin(x) \\\\ &= \left(14 - 7x^2\right)\,sin(x) + 28\,cos(x) \end{align*} \]
